\begin{frame}{Retículas}
    Una retítucla $L$ es un conjunto parcialmente ordenado que tienen dos productos binarios. Si una retícula $L$ tiene
    elementos $0$ y $1$ se puede definir ecuacionalmente como un conjunto con dos elementos distinguidos y dos operaciones
    binarias $\land$ y $\lor$; estas operaciones son tanto asociativas como conmutativas \dots
\end{frame}

% Slide 3: Introducción - Álgebra de Boole
\begin{frame}
    \dots que satisfacen las siguientes identidades.

    \begin{enumerate}
        \item $a \land a = a$, $a \lor a =a$,
        \item $1 \land a = a$, $0 \lor a =a$,
        \item $a \land (b \lor a) = a = (a \land b) \lor a$. 
    \end{enumerate}
\vspace{0.5cm}
    Dadas estas ecuaciones, el orden parcial puede ser recuperado porque $a \leq b$ si y solo si $a \land b = a$ (o bien,
    de manera equivalente $a \lor b = b$).

\vspace{0.5cm}
    El conjunto parcialmente ordenado de todos los subconjuntos de un conjunto dado es siempre un álgebra de booleana.
\end{frame}

\begin{frame}{Álgebra de Heyting}
    Un álgebra de Heyting $H$ (también llamado retícula brouweriana) es una retícula con $0$ y $1$. Dicha retícula tiene,
    para cualquier par de elementos, $x$ y $y$ un exponencial $y^x$. Este exponencial es usualmente denotado como
    $x \Rightarrow y$, que es la mínima cota superior de todos quellos elementos $z \land x \leq y$.
\vspace{0.5cm} 
    \begin{enumerate}
        \item Al realizar la intersección con $x$ el resultado no pasa de $y$.
        \item Cada elemento $z$ que esté por debajo de $y$ (con el ínfimo) debe ser menor o igual al elemento implicación.
    \end{enumerate}
\end{frame}


\begin{frame}{La geometría de la implicación}
    \begin{block}{Definición algebraica}
    La implicación $x \Rightarrow y$ es como un complemento de $x$ respecto a $y$, definido por la adjunción:
    \[ z \leq (x \Rightarrow y) \iff z \wedge x \leq y \]
    \end{block}

    \vfill

    \begin{columns}
        \column{0.5\textwidth}
        \begin{itemize}
            \item $x \Rightarrow y$ es el elemento más alto que hace que $x$ no supere a $y$.
            \item En el diagrama, $y$ siempre está por debajo de $x \Rightarrow y$.
        \end{itemize}

        \column{0.5\textwidth}
        \begin{enumerate}
            \item $x \Rightarrow x = 1$
            \item $x \wedge (x \Rightarrow y) \leq y$
            \item $y \leq (x \Rightarrow y)$
        \end{enumerate}
    \end{columns}
\end{frame}

\begin{frame}{Definición del pseudocomplemento relativo}
    En un álgebra de Heyting $H$, para cada par $x, y \in H$, existe un elemento $x \Rightarrow y$ definido por la propiedad
    
    \begin{block}{Propiedad}
        \[ z \leq (x \Rightarrow y) \iff z \wedge x \leq y \]
    \end{block}
    
    \begin{itemize}
        \item $x \Rightarrow y$ es el máximo del conjunto $\{z \in H : z \wedge x \leq y\}$.
        \item Esto brinda a la retícula de una estructura lógica donde la implicación está siempre definida.
    \end{itemize}
\end{frame}

\begin{frame}{Consecuencias en el orden}
    De la definición se derivan propiedades visuales y algebraicas críticas:
    
    \vfill
    \begin{enumerate}
        \item Ubicación de la consecuencia.
        Como $y \wedge x \leq y$ es siempre verdadera, se sigue que:
        \[ y \leq (x \Rightarrow y) \]
        \textit{En el diagrama, la implicación siempre está por encima de $y$.}
        
        \item Relación con el ínfimo.
        La operación $(\cdot \wedge x)$ y $(x \Rightarrow \cdot)$ forman una relación entre el y y la implicación porque
        estas dos siempre conservan el orden cuando se va de una a la otras.
    \end{enumerate}
    \vfill
    
    \begin{exampleblock}{Intuición}
        Si $y$ es ``verdad'' (está arriba), la implicación $x \Rightarrow y$ debe ser ``más verdad'' o igual.
    \end{exampleblock}
\end{frame}

\begin{frame}{De álgebras Booleanas a Heyting}
    \begin{block}{Teorema}
    Toda álgebra Booleana es un álgebra de Heyting definiendo:
    \[ x \Rightarrow y := \neg x \vee y \]
    \end{block}

    \begin{itemize}
        \item \textbf{Implicación Material:} En lógica clásica, $p \to q \equiv \neg p \vee q$.
        \item \textbf{La Adjunción:} Se cumple que $z \leq (\neg x \vee y) \iff z \wedge x \leq y$.
        \item \textbf{¿Cuál es la diferencia}
        \begin{itemize}
            \item \textbf{Booleana:} El complemento $\neg x$ es global y absoluto ($x \vee \neg x = 1$).
            \item \textbf{Heyting:} La implicación es relativa. Ejemplo: Los abiertos de $\mathbb{R}$ son Heyting pero no Booleanos.
        \end{itemize}
    \end{itemize}
    
    \begin{exampleblock}{Conclusión}
        Las álgebras de Heyting son una generalización de las Booleanas donde el principio del tercero excluido no necesariamente se cumple.
    \end{exampleblock}
\end{frame}

\begin{frame}{Álgebras Booleanas vs. Heyting}
    En un álgebra Booleana, la implicación es \textit{definible}:
    \[ x \Rightarrow y := \neg x \vee y \]
    
    \begin{block}{Teorema de Residuación}
    Para todo $x, y, z$, se cumple la equivalencia:
    \[ z \wedge x \leq y \iff z \leq (\neg x \vee y) \]
    \end{block}

    \begin{itemize}
        \item \textbf{Solo si ($\Rightarrow$):} $z \wedge x \leq (\neg x \vee y) \wedge x = ( \neg x \wedge x) \vee (y \wedge x) = 0 \vee (y \wedge x) \leq y$.
        \item \textbf{Si ($\Leftarrow$):} $z = z \wedge (\neg x \vee x) = (z \wedge \neg x) \vee (z \wedge x) \leq \neg x \vee y$.
    \end{itemize}
\end{frame}

\begin{frame}{La estructura de Heyting}
    Un Álgebra de Heyting es un retículo con $0$ y $1$ donde el funtor $( - \wedge x)$ tiene un \textbf{adjunto derecho} $(x \Rightarrow -)$.
    
    \begin{block}{Identidades Fundamentales}
    \begin{enumerate}
        \item $x \leq (y \Rightarrow (x \wedge y))$ (Unidad)
        \item $y \wedge (y \Rightarrow x) \leq x$ (Counidad / Modus Ponens)
        \item $x \Rightarrow (y \wedge z) = (x \Rightarrow y) \wedge (x \Rightarrow z)$ (Preserva ínfimos)
        \item $(x \vee z) \Rightarrow y = (x \Rightarrow y) \wedge (z \Rightarrow y)$ (Contravariancia)
    \end{enumerate}
    \end{block}
    
     Toda Álgebra de Heyting es un retículo distributivo.
\end{frame}

\begin{frame}{La Negación de Heyting (Pseudocomplemento)}
    La negación se define como la implicación al absurdo:
    \[ \neg x := (x \Rightarrow 0) \]
    
    \begin{alertblock}{Propiedades Diferenciales}
    A diferencia del caso Booleano, en una estructura de Heyting:
    \begin{itemize}
        \item $x \leq \neg \neg x$, pero no siempre $\neg \neg x = x$.
        \item $x \wedge \neg x = 0$, pero no siempre $x \vee \neg x = 1$.
    \end{itemize}
    \end{alertblock}

    \textbf{Ejemplo Geométrico:} En el retículo de abiertos de $\mathbb{R}$, $\neg U$ es el \textit{interior del complemento}. Si $U = (0,1)$, entonces $\neg U = (-\infty, 0) \cup (1, \infty)$. Notar que $U \cup \neg U \neq \mathbb{R}$.
\end{frame}

\begin{frame}{Caracterización Ecuacional de la Implicación}
    Una operación $\Rightarrow$ define una estructura de Heyting si y solo si:
    \begin{itemize}
        \item $x \Rightarrow x = 1$
        \item $x \wedge (x \Rightarrow y) = x \wedge y$
        \item $y \wedge (x \Rightarrow y) = y$
        \item $x \Rightarrow (y \wedge z) = (x \Rightarrow y) \wedge (x \Rightarrow z)$
    \end{itemize}
    \vfill
    \begin{center}
        \textbf{Conclusión:} Un Álgebra de Heyting es Booleana si y solo si $\neg \neg x = x$ para todo $x$.
    \end{center}
\end{frame}
